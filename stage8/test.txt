decl
    integer i,j,arr[10];
    boolean check;
    integer lol;
enddecl
i=2;
lol=1;
arr[lol]=0;
while(i<101)do 
    j=2;
    check=TRUE;
    while(j<i)do
        if(i%j==0)then
            check=FALSE;
        endif;
        j=j+1;
    endwhile;
    if(check)then
        write(i);
    endif;
    i=i+1; 
endwhile;
